\begin{algorithm}
    \caption{Bubble Sort Algorithm}
    \label{alg:bubble}
    \begin{algorithmic}[1]
        \STATE \textbf{Input:} An array $A$ of length $n$
        \STATE \textbf{Output:} Sorted array $A$
        \FOR{$i = 1$ to $n-1$}
            \FOR{$j = 1$ to $n-i$}
                \IF{$A[j] > A[j+1]$}
                    \STATE Swap $A[j]$ and $A[j+1]$
                \ENDIF
            \ENDFOR
        \ENDFOR
        \RETURN $A$
    \end{algorithmic}
\end{algorithm}
